  ---
  Configuration Loading

  Environment variables override config file values, which override defaults.
  - Env var format: RSFGA_<SECTION>__<FIELD> (double underscore as separator)
  - CLI: rsfga --config /path/to/config.yaml

  ---
  All Configuration Options

  server: — HTTP server

  ┌──────────────────────┬─────────┬────────────────────────────────────┬────────────────────────────┐
  │        Field         │ Default │              Env Var               │        Description         │
  ├──────────────────────┼─────────┼────────────────────────────────────┼────────────────────────────┤
  │ host                 │ 0.0.0.0 │ RSFGA_SERVER__HOST                 │ Bind address               │
  ├──────────────────────┼─────────┼────────────────────────────────────┼────────────────────────────┤
  │ port                 │ 8080    │ RSFGA_SERVER__PORT                 │ HTTP port (must be > 0)    │
  ├──────────────────────┼─────────┼────────────────────────────────────┼────────────────────────────┤
  │ request_timeout_secs │ 30      │ RSFGA_SERVER__REQUEST_TIMEOUT_SECS │ Per-request timeout        │
  ├──────────────────────┼─────────┼────────────────────────────────────┼────────────────────────────┤
  │ max_connections      │ 10000   │ RSFGA_SERVER__MAX_CONNECTIONS      │ Max concurrent connections │
  └──────────────────────┴─────────┴────────────────────────────────────┴────────────────────────────┘

  grpc: — gRPC server

  ┌──────────────┬─────────┬──────────────────────────┬───────────────────────────────┐
  │    Field     │ Default │         Env Var          │          Description          │
  ├──────────────┼─────────┼──────────────────────────┼───────────────────────────────┤
  │ enabled      │ true    │ RSFGA_GRPC__ENABLED      │ Enable gRPC server            │
  ├──────────────┼─────────┼──────────────────────────┼───────────────────────────────┤
  │ port         │ 50051   │ RSFGA_GRPC__PORT         │ gRPC port                     │
  ├──────────────┼─────────┼──────────────────────────┼───────────────────────────────┤
  │ reflection   │ true    │ RSFGA_GRPC__REFLECTION   │ gRPC reflection (for grpcurl) │
  ├──────────────┼─────────┼──────────────────────────┼───────────────────────────────┤
  │ health_check │ true    │ RSFGA_GRPC__HEALTH_CHECK │ gRPC health check service     │
  └──────────────┴─────────┴──────────────────────────┴───────────────────────────────┘

  storage: — data backend

  ┌─────────────────────────┬─────────┬────────────────────────────────────────┬───────────────────────────────────────────────┐
  │          Field          │ Default │                Env Var                 │                  Description                  │
  ├─────────────────────────┼─────────┼────────────────────────────────────────┼───────────────────────────────────────────────┤
  │ backend                 │ memory  │ RSFGA_STORAGE__BACKEND                 │ memory, postgres, mysql, cockroachdb, rocksdb │
  ├─────────────────────────┼─────────┼────────────────────────────────────────┼───────────────────────────────────────────────┤
  │ database_url            │ —       │ RSFGA_STORAGE__DATABASE_URL            │ Required for postgres/mysql/cockroachdb       │
  ├─────────────────────────┼─────────┼────────────────────────────────────────┼───────────────────────────────────────────────┤
  │ data_path               │ —       │ RSFGA_STORAGE__DATA_PATH               │ Required for rocksdb (e.g. /data/rsfga)       │
  ├─────────────────────────┼─────────┼────────────────────────────────────────┼───────────────────────────────────────────────┤
  │ pool_size               │ 10      │ RSFGA_STORAGE__POOL_SIZE               │ DB connection pool size                       │
  ├─────────────────────────┼─────────┼────────────────────────────────────────┼───────────────────────────────────────────────┤
  │ connection_timeout_secs │ 5       │ RSFGA_STORAGE__CONNECTION_TIMEOUT_SECS │ DB connect timeout                            │
  └─────────────────────────┴─────────┴────────────────────────────────────────┴───────────────────────────────────────────────┘

  logging:

  ┌───────┬─────────┬──────────────────────┬──────────────────────────────────────────┐
  │ Field │ Default │       Env Var        │               Description                │
  ├───────┼─────────┼──────────────────────┼──────────────────────────────────────────┤
  │ level │ info    │ RSFGA_LOGGING__LEVEL │ trace, debug, info, warn, error          │
  ├───────┼─────────┼──────────────────────┼──────────────────────────────────────────┤
  │ json  │ false   │ RSFGA_LOGGING__JSON  │ JSON format (recommended for production) │
  └───────┴─────────┴──────────────────────┴──────────────────────────────────────────┘

  metrics:

  ┌─────────┬──────────┬────────────────────────┬──────────────────────────────┐
  │  Field  │ Default  │        Env Var         │         Description          │
  ├─────────┼──────────┼────────────────────────┼──────────────────────────────┤
  │ enabled │ true     │ RSFGA_METRICS__ENABLED │ Expose /metrics (Prometheus) │
  ├─────────┼──────────┼────────────────────────┼──────────────────────────────┤
  │ path    │ /metrics │ RSFGA_METRICS__PATH    │ Metrics endpoint path        │
  └─────────┴──────────┴────────────────────────┴──────────────────────────────┘

  tracing: — OpenTelemetry/Jaeger

  ┌─────────────────┬────────────────┬────────────────────────────────┬────────────────────────────┐
  │      Field      │    Default     │            Env Var             │        Description         │
  ├─────────────────┼────────────────┼────────────────────────────────┼────────────────────────────┤
  │ enabled         │ false          │ RSFGA_TRACING__ENABLED         │ Enable distributed tracing │
  ├─────────────────┼────────────────┼────────────────────────────────┼────────────────────────────┤
  │ jaeger_endpoint │ localhost:6831 │ RSFGA_TRACING__JAEGER_ENDPOINT │ Jaeger UDP agent endpoint  │
  ├─────────────────┼────────────────┼────────────────────────────────┼────────────────────────────┤
  │ service_name    │ rsfga          │ RSFGA_TRACING__SERVICE_NAME    │ Trace service label        │
  └─────────────────┴────────────────┴────────────────────────────────┴────────────────────────────┘

  nats: — async writes + RYOW (requires --features nats)

  ┌───────────────────────┬───────────────────────┬───────────────────────────────────┬─────────────────────────────────────────────────────────────┐
  │         Field         │        Default        │              Env Var              │                         Description                         │
  ├───────────────────────┼───────────────────────┼───────────────────────────────────┼─────────────────────────────────────────────────────────────┤
  │ enabled               │ false                 │ RSFGA_NATS__ENABLED               │ Enable NATS integration                                     │
  ├───────────────────────┼───────────────────────┼───────────────────────────────────┼─────────────────────────────────────────────────────────────┤
  │ url                   │ nats://localhost:4222 │ RSFGA_NATS__URL                   │ NATS server URL                                             │
  ├───────────────────────┼───────────────────────┼───────────────────────────────────┼─────────────────────────────────────────────────────────────┤
  │ name                  │ rsfga-api             │ RSFGA_NATS__NAME                  │ Connection name                                             │
  ├───────────────────────┼───────────────────────┼───────────────────────────────────┼─────────────────────────────────────────────────────────────┤
  │ ryow_timeout_secs     │ 30                    │ RSFGA_NATS__RYOW_TIMEOUT_SECS     │ Max wait for write-ticket commit before 504                 │
  ├───────────────────────┼───────────────────────┼───────────────────────────────────┼─────────────────────────────────────────────────────────────┤
  │ write_ticket_ttl_secs │ 60                    │ RSFGA_NATS__WRITE_TICKET_TTL_SECS │ Write ticket validity for RYOW                              │
  ├───────────────────────┼───────────────────────┼───────────────────────────────────┼─────────────────────────────────────────────────────────────┤
  │ write_mode            │ direct                │ RSFGA_NATS__WRITE_MODE            │ direct (sync), nats (async), auto (nats with sync fallback) │
  └───────────────────────┴───────────────────────┴───────────────────────────────────┴─────────────────────────────────────────────────────────────┘

  precompute: — Valkey/Redis cache (requires --features precompute)

  ┌──────────────────┬────────────────────────┬────────────────────────────────────┬──────────────────────────────────────────────────────────────────────┐
  │      Field       │        Default         │              Env Var               │                             Description                              │
  ├──────────────────┼────────────────────────┼────────────────────────────────────┼──────────────────────────────────────────────────────────────────────┤
  │ enabled          │ false                  │ RSFGA_PRECOMPUTE__ENABLED          │ Enable precomputed check cache                                       │
  ├──────────────────┼────────────────────────┼────────────────────────────────────┼──────────────────────────────────────────────────────────────────────┤
  │ valkey_url       │ redis://localhost:6379 │ RSFGA_PRECOMPUTE__VALKEY_URL       │ Supports redis://, rediss:// (TLS), redis+sentinel://, redis+unix:// │
  ├──────────────────┼────────────────────────┼────────────────────────────────────┼──────────────────────────────────────────────────────────────────────┤
  │ result_ttl_secs  │ 300                    │ RSFGA_PRECOMPUTE__RESULT_TTL_SECS  │ TTL for cached check results (must be > 0)                           │
  ├──────────────────┼────────────────────────┼────────────────────────────────────┼──────────────────────────────────────────────────────────────────────┤
  │ hotpath_ttl_secs │ 3600                   │ RSFGA_PRECOMPUTE__HOTPATH_TTL_SECS │ TTL for hot-path registry entries (must be > 0)                      │
  └──────────────────┴────────────────────────┴────────────────────────────────────┴──────────────────────────────────────────────────────────────────────┘

  ---
  Full config.yaml example

  server:
    host: "0.0.0.0"
    port: 8080
    request_timeout_secs: 30
    max_connections: 10000

  grpc:
    enabled: true
    port: 50051
    reflection: true
    health_check: true

  storage:
    backend: postgres
    database_url: "postgres://user:password@localhost:5432/rsfga"
    pool_size: 20
    connection_timeout_secs: 5

  logging:
    level: info
    json: true

  metrics:
    enabled: true
    path: /metrics

  tracing:
    enabled: false
    jaeger_endpoint: "localhost:6831"
    service_name: rsfga

  nats:
    enabled: false
    url: "nats://localhost:4222"
    name: "rsfga-api"
    ryow_timeout_secs: 30
    write_ticket_ttl_secs: 60
    write_mode: "direct"

  precompute:
    enabled: false
    valkey_url: "redis://localhost:6379"
    result_ttl_secs: 300
    hotpath_ttl_secs: 3600

  ---
  Startup validation (fatal errors)

  ┌───────────────────────────────────────────────┬─────────────────────────────────────────────────────────────────┐
  │                    Config                     │                              Rule                               │
  ├───────────────────────────────────────────────┼─────────────────────────────────────────────────────────────────┤
  │ server.port                                   │ Must be > 0                                                     │
  ├───────────────────────────────────────────────┼─────────────────────────────────────────────────────────────────┤
  │ storage.backend                               │ Must be one of the 5 listed values                              │
  ├───────────────────────────────────────────────┼─────────────────────────────────────────────────────────────────┤
  │ storage.database_url                          │ Required (non-empty) when backend is postgres/mysql/cockroachdb │
  ├───────────────────────────────────────────────┼─────────────────────────────────────────────────────────────────┤
  │ storage.data_path                             │ Required (non-empty) when backend is rocksdb                    │
  ├───────────────────────────────────────────────┼─────────────────────────────────────────────────────────────────┤
  │ logging.level                                 │ Must be a valid level name                                      │
  ├───────────────────────────────────────────────┼─────────────────────────────────────────────────────────────────┤
  │ nats.write_mode                               │ Must be nats/direct/auto (only checked when nats.enabled=true)  │
  ├───────────────────────────────────────────────┼─────────────────────────────────────────────────────────────────┤
  │ precompute.valkey_url                         │ Non-empty when precompute.enabled=true                          │
  ├───────────────────────────────────────────────┼─────────────────────────────────────────────────────────────────┤
  │ precompute.result_ttl_secs / hotpath_ttl_secs │ Must be > 0 when precompute.enabled=true                        │
  └───────────────────────────────────────────────┴─────────────────────────────────────────────────────────────────┘
